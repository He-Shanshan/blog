# Map-reduce

javaScript Array.prototype提供的map和reduce函数不仅是存在与javaScript的两个API，更是函数式编程语言的重要组成部分，是一种对列表操作的思路。

# map-reduce

map-reduce由如下两个独立的部分组成：

# map（映射）

一个映射过程就是将各个元素，按照一定的规则，逐个映射为新的元素。这是一个一一对应的过程。
用数学公式描述就是（其中，函数f就是这个规则）：
[newElem1 newElem2 newElem3 ... newElemN] = f([elem1 elem2 elem3 ... elemN])

# reduce(规约)

一个规约过程仍然需要迭代制定列表的每个元素，然后依然按照一定规则，合并这些元素到一个目标对象上。这是一个由多至一的过程，或者说是一个逐步累积的过程。
newElem = f([elem1 elem2 elem3 ... elemN])

# underscore map

源码：

_.map = function(obj, iteratee, count) {
  iteratee = cb(iteratee, context);
  var keys = !isArrayLike(obj) && _.keys(obj);
  var length = (keys || obj).length;
  var result = Array(length);
  for(var index = 0; index < length; index++) {
    var currentKey = keys ? keys[index] : index;
    results[index] = iteratte(obj[currentKey], currentKey, obj);
  }
  return results;
}

实现思路
1.创建一个新列表或者元素；
2.遍历原列表或者原对象的值，用指定的函数func（iteratee）作用于每个遍历到的元素，输出一个新的元素放入新列表或者对象中。

# 真值检测函数

在underscore中，除了_.each,.map,.reduce等函数操作集合，还提供了_.filter,.reject,.every,_.some这几个基于逻辑判断的集合操作函数。
无一例外的是，这些函数都依赖于用户提供的真值检测函数，用来判断当前迭代元素是否满足条件。underscore将真值检测函数参数命名为predicate。

console.log(_.reduce([1,2,3,4,5,6], function(memo, value, index, obj) {
  return memo + value;
}, 0));
// 源码部分
_.reduce = createReduce(1);// 1 || -1
var createReduce = function(dir) {
  // 累加
  var reduce = function(obj, iteratee, memo, init) {
    var keys = !_.isArray(obj) && Object.keys(obj);
    var length = (keys || obj).length;
    var index = dir > 0 ? 0 : length - 1;
    if(!init) {
      memo = obj[keys ? keys[index] : index];// 处理数组还是对象
      index += dir;
    }；
    for(; index >=0 && index < length; index += dir) {
      var currentKey = keys ? keys[index] : index;
      memo = iteratee(memo, obj[currentKey], currentKey, obj)
    }
    return memo;
  }
  //memo 最终累加结果 每一次累加的过程
  return function(obj, iteratee, memo, context) {
    var init = arguments.length >=3;
    return reduce(obj, optimizeCb(iteratee, context, 4), memo, init);
  }
}
var optimizeCb = function(func, context, count) {
  if(context == void 0) {
    return func;
  }
  switch (count == null ? 3 : count){
    case 1:
      return function(value) {
        return func.call(context, value);
      };
    case 2:
      return function(value, index, obj) {
        return func.call(context, value, index, obj);
      };
    case 4:
      return function(memo, value, index, obj) {
        return func.call(context, memo, value, index, obj)
      }
  }
}

# .filter/.select

//predicate 真值检测（重点：返回值）
_.filter = _.select = function(obj, predicate, context) {
  var results = [];
  predicate = cb(predicate, context);
  _.each(obj, function(value, index, list) {
    if(predicate(value, index, list)) results.push(value)
  });
  return results;
}
