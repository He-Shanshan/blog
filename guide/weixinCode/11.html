# 部署后端

# 云开发配置

在小程序开发工具中，点击云开发按钮，填写对应的环境名称，环境ID，点击确定等待处理完成。

# 云函数

# 创建一个云函数

1.在根目录创建一个名为cloudfunctions的文件夹。
2.配置project.config.json

{
  "cloudfunctionRoot": "cloudfunctions/",
}
1
2
3

保存，这时候cloudfunctions文件夹后面会显示出当前云服务名称。
3.右键cloudfunctions文件夹，点击新建Node.js云函数，输入新建云函数名称如return,此时点击云开发，看到云函数里面存在一个名称为return的云函数。
此时我们修改return文件夹下的index.js文件，使其返回的数据多个字段sum

// 云函数入口文件
const cloud = require('wx-server-sdk')

cloud.init()

// 云函数入口函数
exports.main = async (event, context) => {
  const wxContext = cloud.getWXContext()

  return {
    sum:event.a + event.b,// 新增返回字段
    event,
    openid: wxContext.OPENID,
    appid: wxContext.APPID,
    unionid: wxContext.UNIONID,
  }
}

删除多条需要在Serve端进行（云函数）操作

// 云函数入口文件
const cloud = require('wx-server-sdk')
// 初始化
const db = wx.cloud.database()
cloud.init()

// 云函数入口函数
exports.main = async (event, context) => {
  try{
    return await db.collection('todos').where({
      done:true
    }).remove()
  } catch(e){
    console.error(e)
  }
}
