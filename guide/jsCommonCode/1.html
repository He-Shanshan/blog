# 数组去重

# 双层循环

function unique(arr) {
  var res = [];
  for (var i = 0; i < arr.length; i++) {
    for (var j = 0; j < res.length; j++) {
      if (arr[i] === res[j]) {
        break;
      }
    }
    // 如果查找到最后一位认为
    if (j === res.length) {
      res.push(arr[i])
    }  
  }
  return res;
}

# 排序后去重

如果我们对一个已经排好序的数组去重，这种方法效率肯定高于使用 indexOf。

function unique(arr) {
  var uniqueArr = [];
  var sortArr = arr.concat().sort();//排序并且不更改原数组的顺序
  for (var i = 0; i < sortArr.length; i++) {
    if (!i || sortArr[i] != sortArr[i - 1]) {//第一个可以直接push进去,每个元素都与前一个元素做比较
      uniqueArr.push(sortArr[i])
    }
  }
  return uniqueArr;

}

# 仿underscore的去重写法

尽管 unqique 已经可以试下去重功能，但是为了让这个 API 更加强大，我们来考虑一个需求：

新需求：字母的大小写视为一致，比如'a'和'A'，保留一个就可以了！

虽然我们可以先处理数组中的所有数据，比如将所有的字母转成小写，然后再传入unique函数，但是有没有方法可以省掉处理数组的这一遍循环，直接就在去重的循环中做呢？让我们去完成这个需求：

var array = [1, 1, 'a', 'A', 2, 2];;
function unique(arr, isSorted, iteratee) {
  var uniqueArr = [];
  for (var i = 0; i < arr.length; i++) {
    var computed = iteratee ? iteratee(arr[i], i, arr) : arr[i];
    var seen;
    if (isSorted) {
      if (!i || computed != seen) {
        uniqueArr.push(arr[i]);
        seen = computed;
      }
    } else if (iteratee) {
      if (uniqueArr.indexOf(computed) === -1) {
        uniqueArr.push(arr[i])
      }
    } else {
      if (uniqueArr.indexOf(arr[i]) === -1) {
        uniqueArr.push(arr[i])
      }
    }
  }
  return uniqueArr;

}

  return typeof item == 'string' ? item.toLowerCase() : item
}))

# filter

var array = [1, 1, 'a', 'A', 2, 2];;
function unique(arr) {
  var res = arr.filter(function (item, index, arr) {
    return arr.indexOf(item) === index
  })
  return res;

# ES6

随着 ES6 的到来，去重的方法又有了进展，比如我们可以使用 Set 和 Map 数据结构，以 Set 为例，ES6 提供了新的数据结构 Set。它类似于数组，但是成员的值都是唯一的，没有重复的值。

# set

var array = [1, 2, 1, 1, '1'];

function unique(array) {
   return Array.from(new Set(array));
}

console.log(unique(array)); // [1, 2, "1"]

    function unique(array) {
        var res = [];
        for (var i = 0, len = array.length; i < len; i++) {
            var current = array[i];

            if (Array.isArray(array[i])) {
                res.push(unique(array[i]))
            }
            else if (res.indexOf(current) === -1) {
                res.push(current)
            }

        }
        return res;
    }
